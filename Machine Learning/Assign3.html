#run k-NN in a loop with different values of k (starting from k=5) and compute the testing misclassification rate
from sklearn.neighbors import KNeighborsClassifier
smallest = 1
for k in range(5, X_train.shape[0]):
    kNNObj = KNeighborsClassifier(n_neighbors=k)
    kNNObj.fit(X_train, y_train)
    misclass =(1-kNNObj.score(X_test,y_test))
    if misclass < smallest:
        smallest=misclass
        bestk = k
    
print("The best value for k is " , bestk)
print("with a misclassification rate of " ,smallest)

#Prediction
kNNObj = KNeighborsClassifier(n_neighbors=bestk)
kNNObj.fit(X_train, y_train)
y_pred_knn= kNNObj.predict(X_test)

9) Model 5: Fit Random Forest model. For Random Forest, you need to specify the number of trees (n_estimators). In order to figure that out, run Random Forest in a loop with different values of n_estimators (starting from 10) and compute the testing misclassification rate. Plot a chart with n_estimators on X-axis and testing error on the Y-axis. What is the lowest value of testing error and corresponding value of n_estimators?

In [10]:
from sklearn.ensemble import RandomForestClassifier
est = []
err = []
for n in range(10, X_train.shape[0]):
    RFObj = RandomForestClassifier(n_estimators = n, criterion='entropy')
    RFObj.fit(X_train,y_train)
    est.append(n)
    err.append(1-RFObj.score(X_test,y_test))
    

print("The smallest misclassication error is: ", min(err))
print("Corresponding to n_estimators = ", est[err.index(min(err))])

# Plot 
plt.scatter(est,err)
plt.show()
